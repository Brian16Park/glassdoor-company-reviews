    )
}

f1_score <- function(actual, predicted) {
  tp <- sum(actual == "POSITIVE" & predicted == "POSITIVE")
  fp <- sum(actual == "NEGATIVE" & predicted == "POSITIVE")
  fn <- sum(actual == "POSITIVE" & predicted == "NEGATIVE")
  
  precision <- ifelse((tp + fp) == 0, 0, tp / (tp + fp))
  recall <- ifelse((tp + fn) == 0, 0, tp / (tp + fn))
  
  if (precision + recall == 0) return(0)
  
  2 * precision * recall / (precision + recall)
}

evaluate_model <- function(prob, actual, thresholds = seq(0.1, 1.0, by = 0.05)) {
  f1_results <- tibble(threshold = thresholds) %>%
    mutate(
      F1 = purrr::map_dbl(threshold, ~{
        pred_class <- ifelse(prob > .x, "POSITIVE", "NEGATIVE")
        f1_score(actual, pred_class)
      })
    )
  
  best_threshold <- f1_results %>%
    slice_max(F1, n = 1, with_ties = FALSE) %>%
    pull(threshold)
  
  pred_class <- factor(
    ifelse(prob > best_threshold, "POSITIVE", "NEGATIVE"),
    levels = c("NEGATIVE", "POSITIVE")
  )
  
  cm <- confusionMatrix(pred_class, actual)
  roc_obj <- roc(actual, prob, levels = c("NEGATIVE", "POSITIVE"), direction = "<")
  
  list(
    f1_results = f1_results,
    best_threshold = best_threshold,
    pred_class = pred_class,
    confusion_matrix = cm,
    roc = roc_obj,
    auc = as.numeric(auc(roc_obj))
  )
}

extract_glm_importance <- function(model, features_name, model_name, top_n = 10) {
  broom::tidy(model) %>%
    dplyr::filter(term != "(Intercept)") %>%
    dplyr::mutate(
      Features = features_name,
      Model = model_name,
      Importance = abs(estimate),
      Direction = ifelse(estimate >= 0, "Positive", "Negative")
    ) %>%
    dplyr::arrange(desc(Importance)) %>%
    dplyr::slice_head(n = top_n) %>%
    dplyr::select(Features, Model, term, estimate, Importance, Direction)
}

extract_glmnet_importance <- function(cv_model, features_name, model_name, s = "lambda.min", top_n = 10) {
  coef_df <- as.matrix(coef(cv_model, s = s)) %>%
    as.data.frame() %>%
    tibble::rownames_to_column("term")
  
  names(coef_df)[2] <- "estimate"
  
  coef_df %>%
    dplyr::filter(term != "(Intercept)", estimate != 0) %>%
    dplyr::mutate(
      Features = features_name,
      Model = model_name,
      Importance = abs(estimate),
      Direction = ifelse(estimate >= 0, "Positive", "Negative")
    ) %>%
    dplyr::arrange(desc(Importance)) %>%
    dplyr::slice_head(n = top_n) %>%
    dplyr::select(Features, Model, term, estimate, Importance, Direction)
}

extract_rf_importance <- function(rf_model_tuned, features_name, model_name, top_n = 10) {
  imp <- importance(rf_model_tuned$finalModel, type = 1)
  
  data.frame(
    term = rownames(imp),
    Importance = imp[, 1],
    stringsAsFactors = FALSE
  ) %>%
    dplyr::mutate(
      Features = features_name,
      Model = model_name,
      estimate = Importance,
      Direction = "RF Importance"
    ) %>%
    dplyr::arrange(desc(Importance)) %>%
    dplyr::slice_head(n = top_n) %>%
    dplyr::select(Features, Model, term, estimate, Importance, Direction)
}

5. Fit Logistic Sentiment-Addons

Why Add Sentiment Features?

While structured features (ratings, flags, word counts) capture explicit signals in employee reviews, they may miss the underlying tone and nuance in textual feedback. Sentiment analysis helps quantify this tone by transforming free-text fields into numerical features that reflect positivity, negativity, and variability across different parts of a review.

We incorporated three sentiment models:

  • VADER: A rule-based sentiment analyzer designed for short, informal text. It produces a compound score capturing overall sentiment intensity.
  • DistilBERT (distilbert-base-uncased-finetuned-sst-2-english): A lightweight transformer model fine-tuned for sentiment classification, providing strong contextual understanding with lower computational cost.
  • RoBERTa (cardiffnlp/twitter-roberta-base-sentiment-latest): A more powerful transformer model trained on large-scale data, capable of capturing subtle linguistic nuances and contextual sentiment shifts.

By adding these features, we aim to test whether textual tone provides additional predictive power beyond structured variables.

for (name in names(datasets)) {
  df_current <- datasets[[name]]
  
  train_df <- df_current[train_index, ]
  test_df  <- df_current[-train_index, ]

  x_test  <- model.matrix(flags_recommend_frend ~ ., data = test_df)[, -1]
  y_train <- ifelse(train_df$flags_recommend_frend == "POSITIVE", 1, 0)
  
  set.seed(123)
  model_logit <- glm(
    flags_recommend_frend ~ .,
    data = train_df,
    family = binomial
  )
  prob_logit <- predict(model_logit, newdata = test_df, type = "response")
  eval_logit <- evaluate_model(prob_logit, test_df$flags_recommend_frend)
  
  set.seed(123)
  cv_lasso <- cv.glmnet(x_train, y_train, family = "binomial", alpha = 1)
  prob_lasso <- as.numeric(
    predict(cv_lasso, newx = x_test, s = "lambda.min", type = "response")
  )
  eval_lasso <- evaluate_model(prob_lasso, test_df$flags_recommend_frend)
  
  set.seed(123)
  cv_ridge <- cv.glmnet(x_train, y_train, family = "binomial", alpha = 0)
  prob_ridge <- as.numeric(
    predict(cv_ridge, newx = x_test, s = "lambda.min", type = "response")
  )
  eval_ridge <- evaluate_model(prob_ridge, test_df$flags_recommend_frend)
  
  set.seed(123)
  cv_enet <- cv.glmnet(x_train, y_train, family = "binomial", alpha = 0.5)
  prob_enet <- as.numeric(
    predict(cv_enet, newx = x_test, s = "lambda.min", type = "response")
  )
  eval_enet <- evaluate_model(prob_enet, test_df$flags_recommend_frend)
  
  logit_results[[name]] <- list(

    logit_eval = eval_logit,

    enet_eval = eval_enet

rf_results <- list()

for (name in names(datasets)) {
  df_current <- datasets[[name]]
  
  train_df <- df_current[train_index, ]
  test_df  <- df_current[-train_index, ]
  
  p <- ncol(train_df) - 1
  
  set.seed(123)
  rf_model <- train(
    flags_recommend_frend ~ .,
    data = train_df,
    method = "rf",
    trControl = trainControl(method = "cv", number = 10),
    tuneGrid = expand.grid(mtry = 1:p),
    ntree = 500,
    importance = TRUE
  )
  
  bestTune <- rf_model$bestTune
  
  set.seed(123)
  rf_model_tuned <- train(
    flags_recommend_frend ~ .,
    data = train_df,
    method = "rf",
    trControl = trainControl(
      method = "cv",
      number = 10,
      classProbs = TRUE,
      summaryFunction = twoClassSummary
    ),
    tuneGrid = bestTune,
    ntree = 500,
    metric = "ROC",
    importance = TRUE
  )
  
  prob_rf <- predict(rf_model_tuned, newdata = test_df, type = "prob")[, "POSITIVE"]
  eval_rf <- evaluate_model(prob_rf, test_df$flags_recommend_frend)
  
  rf_results[[name]] <- list(
    best_mtry = bestTune$mtry,
    rf_model = rf_model,
    rf_model_tuned = rf_model_tuned,
    eval = eval_rf
  )
  
  rf_importance[[paste0(name, "_rf")]] <- extract_rf_importance(
    rf_model_tuned, name, "Random Forest"
  )
}
